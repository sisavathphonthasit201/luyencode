using System;

namespace bai1
{
    class Program
    {
        static void Main(string[] args)
        {
            var s = Console.ReadLine(); //s="1 2"
            var s1 = s.Split(' '); //s1[0]="1" s1[1]="2"

            int c = Convert.ToInt32(s1[2]);
            int d = a + b + c;
            Console.WriteLine(d);
        }
    }
