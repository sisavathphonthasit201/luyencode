using System;

namespace bai1
{
    class Program
    {
        static void Main(string[] args)
        {
            var s = Console.ReadLine(); //s="1 2"
            
            double r = Convert.ToDouble(s);

            double chuvi = 2 * 3.14 * r;
            double dientich = 3.14 * r * r;
            Console.WriteLine("{0} {1}",chuvi.ToString("0.000"), dientich.ToString("0.000"));
            
        }
    }
}
